Базовый пример подключения и использования

<!doctype html>
<html>
<head>
    <!-- Зависимости: jQuery и Handlebars -->
    <script src="https://ajax.googleapis.com/ajax/libs/jquery/1.12.4/jquery.min.js"></script>
    <script src="https://cdnjs.cloudflare.com/ajax/libs/handlebars.js/4.0.5/handlebars.min.js"></script>

    <!-- Плагин: javascript и CSS -->
    <script src="jquery.goodpopup/js/script.min.js"></script>
    <link href="jquery.goodpopup/css/style.min.css" rel="stylesheet" type="text/css"/>
        
    <!-- Стилизуйте внутренний контент попапа как хотите -->
    <style>
        .popup {
            background: #fff;
            padding: 10px 40px;
            text-align: center;
        }
        .popup__header {
            font-size: 2em;
        }
    </style>
</head>
<body>
    <!-- Поставьте кнопку, по клику на которую будет открыт попап -->
    <button id="button">Открыть попап</button>

    <!-- Создайте разметку внутреннего контента попапа в формате handlebars-шаблона, используя уникальный id -->
    <script id="demo-popup" type="text/x-handlebars-template">
        <div class="popup">
            <h1 class="popup__header">Привет, я попап!</h1>
        </div>
    </script>

    <!-- Повесьте обработчик клика на кнопку. Обратитесь к попапу по его id и откройте его, используя метод open() -->
    <script>
        $(function() {
            $("#button").on("click", function() {
                $.goodpopup("demo-popup").open();
            });
        });
    </script>
</body>
</html>
